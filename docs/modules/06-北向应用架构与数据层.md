# 北向应用架构与数据层 — 模块设计文档

> **模块名称**: 北向应用架构与数据层 (Northbound App Architecture & Data Layer)  
> **所属项目**: 鸿蒙虚拟数字宠物 (HarmoPet)  
> **模块版本**: v1.0  
> **更新日期**: 2026-04-15  
> **技术栈**: ArkTS + ArkUI / OpenHarmony SDK API 12+  

---

## 1. 模块概述

### 1.1 模块定位

本模块定义了 HarmoPet 北向应用的**整体软件架构**、**代码分层规范**、**数据持久化方案**以及**页面路由体系**。这是所有其他模块（核心宠物、AI智能、社交系统等）的运行容器和基础设施。

### 1.2 技术选型总览

| 领域 | 技术方案 | 版本/说明 |
|------|----------|-----------|
| 开发语言 | ArkTS (TypeScript超集) | 基于OpenHarmony SDK |
| UI框架 | ArkUI (声明式UI) | @Component装饰器, @Builder |
| 运行环境 | OpenHarmony / SwanLinkOS 5.0 | API 12+ |
| 状态管理 | @State / @Prop / @Link / @Provide / @Watch | ArkUI响应式机制 |
| 轻量持久化 | PersistentStorage (@ohos.data.preferences) | KV键值对存储 |
| 结构化存储 | RDB 关系型数据库 (@ohos.data.relationalStore) | SQLite引擎 |
| 网络通信 | @ohos.net.http / WebSocket | REST + 实时消息 |
| 动画系统 | Animator / Canvas / Lottie | 宠物精灵帧动画 |
| 音频处理 | @ohos.multimedia.media | 录音与播放 |
| 分布式能力 | @ohos.distributedDeviceManager | 双端设备发现(可选) |

```typescript
@Entry
@Component
struct AppEntry {
  aboutToAppear(): void {
    this.checkFirstLaunch();
  }
  
  private async checkFirstLaunch(): Promise<void> {
    const adoptionService = AdoptionService.getInstance();
    
    if (!adoptionService.isAdopted()) {
      // 未领养 → 进入领养流程
      router.replaceUrl({ url: 'pages/AdoptPage' });
    } else {
      // 已领养 → 初始化各服务 → 进入主页
      await this.initializeServices();
      router.replaceUrl({ url: 'pages/MainPetPage' });
    }
  }
  
  private async initializeServices(): Promise<void> {
    // 按依赖顺序初始化各服务
    await SettingsService.getInstance().loadFromStorage();       // 1. 加载设置
    await PetAttributeService.getInstance().initialize();        // 2. 属性服务(RDB恢复)
    await PersonalityService.getInstance().initialize();        // 3. 性格服务(RDB恢复)
    await AchievementService.getInstance().initialize();         // 4. 成就服务(RDB恢复)
    PetAttributeService.getInstance().startDecayTimer();        // 5. 启动属性衰减定时器
    
    // 尝试连接AI后端(异步，不阻塞启动)
    AIService.getInstance().connect(SettingsService.getConfig().ai.apiBaseUrl);
  }
}
```

```typescript
// 文件路径: pages/MainPetPage.ets

@Component
export struct MainPetPage {
  // ---------- 状态 ----------
  @State petAttributes: PetAttributes = getDefaultAttributes();
  @State currentEmotion: EmotionType = 'calm';
  @State bubbleText: string = '';
  @State showBubble: boolean = false;
  
  // ---------- 生命周期 ----------
  aboutToAppear(): void {
    // 订阅属性变化
    PetAttributeService.getInstance()
      .onAttributeChange()
      .subscribe((change) => {
        this.petAttributes = PetAttributeService.getInstance().getAllAttributes();
      });
    
    // 订阅情绪变化
    EmotionService.getInstance()
      .onEmotionChange()
      .subscribe((state) => {
        this.currentEmotion = state.type;
      });
  }
  
  // ---------- 交互处理 ----------
  private onQuickAction(action: string): void {
    switch (action) {
      case 'feed': router.pushUrl({ url: 'pages/InteractPage', params: { tab: 'feed' } }); break;
      case 'chat': router.pushUrl({ url: 'pages/ChatPage' }); break;
      case 'profile': router.pushUrl({ url: 'pages/ProfilePage' }); break;
      case 'settings': router.pushUrl({ url: 'pages/SettingsPage' }); break;
      default: break;
    }
  }
  
  build() {
    Column() {
      // 顶部属性栏
      AttributeBar({ attributes: $petAttributes })
      
      // 中间宠物精灵区域 (弹性填充)
      PetSprite({
        currentAction: /* 由AnimationService驱动 */,
        emotion: $currentEmotion,
        scale: 1.0,
        onSpriteTouch: (e) => this.onPetTouched(e),
      })
      
      // 对话气泡
      if (this.showBubble) {
        ChatBubble({ text: this.bubbleText })
      }
      
      // 快捷操作栏
      QuickActionBar({ onAction: (action) => this.onQuickAction(action) })
    }
  }
}
```

#### 聊天状态管理

```typescript
// 文件路径: pages/ChatPage.ets

interface ChatMessageItem {
  id: string;
  role: 'user' | 'pet';
  content: string;
  timestamp: number;
  audioData?: string;          // TTS音频(仅pet角色)
  isPlayingAudio?: boolean;
}

@Component
export struct ChatPage {
  @State messages: ChatMessageItem[] = [];
  @State inputText: string = '';
  @State isLoading: boolean = false;
  @State isRecording: boolean = false;
  
  private chatHistory: ChatMessageItem[] = [];
  
  // ---------- 发送文字消息 ----------
  private async sendTextMessage(): Promise<void> {
    if (!this.inputText.trim()) return;
    
    const userMsg: ChatMessageItem = {
      id: generateId(),
      role: 'user',
      content: this.inputText,
      timestamp: Date.now(),
    };
    this.messages.push(userMsg);
    const userInput = this.inputText;
    this.inputText = '';
    this.isLoading = true;
    
    try {
      // 调用AI对话接口
      const aiService = AIService.getInstance();
      const response = await aiService.chat({
        message: userInput,
        sessionId: this.getSessionId(),
        petState: this.buildPetState(),
        needVoice: SettingsService.getConfig().sound.ttsVolume > 0,
      });
      
      const petMsg: ChatMessageItem = {
        id: generateId(),
        role: 'pet',
        content: response.reply,
        timestamp: Date.now(),
        audioData: response.audioData,
      };
      this.messages.push(petMsg);
      
      // 如果有音频，自动播放
      if (response.audioData) {
        VoiceService.getInstance().playAudio(response.audioData, response.audioFormat);
      }
    } catch (error) {
      // AI不可用时使用预设回复
      const fallback = FallbackResponseManager.getRandomResponse('default');
      this.messages.push({
        id: generateId(), role: 'pet', content: fallback, timestamp: Date.now(),
      });
    } finally {
      this.isLoading = false;
    }
  }
  
  // ---------- 语音输入 ----------
  private async toggleVoiceInput(): Promise<void> {
    if (!this.isRecording) {
      this.isRecording = true;
      const taskId = await VoiceService.getInstance().startRecording();
      // UI显示录音中动画...
    } else {
      this.isRecording = false;
      const result = await VoiceService.getInstance().stopAndSend(this.currentTaskId);
      this.inputText = result.recognizedText;
      if (result.recognizedText) {
        this.sendTextMessage();  // 自动发送识别结果
      }
    }
  }
}
```

### 5.2 RDB数据库Schema定义

```typescript
// 文件路径: data/RDBSchema.ets

/** 
 * RDB数据库管理器
 * 数据库名: harmopet.db
 */

const DATABASE_NAME = 'harmopet.db';
const DATABASE_VERSION = 1;

// ========== 表: interaction_log (互动日志) ==========
// 用途: 记录每次用户互动操作（赛题"永久存储互动内容"核心表）
const TABLE_INTERACTION_LOG = `
CREATE TABLE IF NOT EXISTS interaction_log (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  interaction_type TEXT NOT NULL,           -- 互动类型: feed/pet/play/train/bath/sleep
  params TEXT,                              -- 附加参数(JSON): {"foodId":"fish"}
  attribute_changes TEXT,                   -- 属性变化(JSON): {"health":12,"mood":5}
  bubble_text TEXT,                         -- 宠物气泡回复
  created_at INTEGER NOT NULL,              -- Unix时间戳(ms)
  synced_to_ai INTEGER DEFAULT 0            -- 是否已同步到AI后端(0/1)
);
CREATE INDEX IF NOT EXISTS idx_interaction_time ON interaction_log(created_at DESC);
CREATE INDEX IF NOT EXISTS idx_interaction_type ON interaction_log(interaction_type);
`;

// ========== 表: chat_history (对话历史) ==========
const TABLE_CHAT_HISTORY = `
CREATE TABLE IF NOT EXISTS chat_history (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  session_id TEXT NOT NULL,                -- 会话ID
  role TEXT NOT NULL,                      -- 角色: user/pet
  content TEXT NOT NULL,                   -- 消息内容
  audio_path TEXT,                         -- TTS音频本地路径(可选)
  emotion_at_time TEXT,                    -- 对话时的宠物情绪
  created_at INTEGER NOT NULL,
  synced_to_ai INTEGER DEFAULT 0
);
CREATE INDEX IF NOT EXISTS idx_chat_session ON chat_history(session_id, created_at ASC);
`;

// ========== 表: dream_journal (梦境日记) ==========
const TABLE_DREAM_JOURNAL = `
CREATE TABLE IF NOT EXISTS dream_journal (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  date TEXT NOT NULL,                      -- 日期 YYYY-MM-DD
  dream_type TEXT NOT NULL,               -- 类型: happy/recall/fantasy/anxious/food
  title TEXT,                             -- 梦境标题
  content TEXT NOT NULL,                  -- 梦境描述
  illustration_url TEXT,                  -- AI绘图URL(P2)
  emotion_before TEXT,                    -- 入睡时情绪
  created_at INTEGER NOT NULL
);
```

### 6.2 服务初始化顺序

```typescript
// 文件路径: entry/src/main/ets/ServiceInitializer.ets

export class ServiceInitializer {
  static async initializeAll(context: Context): Promise<void> {
    // Phase 1: 基础设施 (必须成功)
    await RDBHelper.getInstance().initialize(context);     // 1. RDB数据库
    await PrefHelper.getInstance().initialize(context);     // 2. Preferences
    await SettingsService.getInstance().loadFromStorage();  // 3. 加载设置
    
    // Phase 2: 核心业务服务
    await PetAttributeService.getInstance().initialize();   // 4. 属性服务
    await PersonalityService.getInstance().initialize();    // 5. 性格服务
    await AchievementService.getInstance().initialize();    // 6. 成就服务
    
    // Phase 3: 启动定时任务
    PetAttributeService.getInstance().startDecayTimer();   // 7. 属性衰减
    
    // Phase 4: 外部服务(异步，不阻塞)
    this.connectExternalServices();                        // 8. AI/N外设连接(异步)
  }
  
  private static connectExternalServices(): void {
    const config = SettingsService.getConfig();
    
    // 连接AI后端
    AIService.getInstance()
      .connect(config.ai.apiBaseUrl)
      .then(success => {
        if (success) console.info('[Init] AI service connected');
        else console.warn('[Init] AI service unavailable, using fallback');
      });
    
    // 尝试连接南向设备
    SouthDeviceService.getInstance()
      .connect()
      .then(success => {
        if (success) console.info('[Init] South device connected');
        else console.info('[Init] No south device, using mock data');
      });
    
    // 尝试连接全志端
    HoloSyncService.getInstance()
      .detectAndInitialize();
  }
}
```
