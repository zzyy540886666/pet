# AI服务后端架构 — 模块设计文档

> **模块名称**: AI服务后端架构 (AI Service Backend Architecture)  
> **所属项目**: 鸿蒙虚拟数字宠物 (HarmoPet)  
> **模块版本**: v1.0  
> **更新日期**: 2026-04-15  
> **技术栈**: Python 3.10+ / FastAPI / Whisper / Edge-TTS / ChromaDB / SQLite

---

## 1. 模块概述

### 1.1 模块定位

AI服务后端是 HarmoPet 的**智能核心**，以 Python FastAPI 提供 RESTful API 服务。北向 ArkTS 应用通过 HTTP/WebSocket 调用本服务，实现对话、语音、记忆、情绪、梦境等全部AI能力。该后端可部署在本地笔记本或云服务器上。

### 1.2 技术选型

| 领域 | 技术方案 | 版本 | 说明 |
|------|----------|------|------|
| Web框架 | FastAPI | ≥0.100.0 | 高性能异步API框架 |
| LLM接口 | OpenAI API兼容 | - | 支持OpenAI/DeepSeek/通义千问等 |
| ASR(语音识别) | Whisper (openai-whisper) | latest | 本地离线语音识别 |
| TTS(语音合成) | edge-tts | ≥6.1.0 | 微软Edge在线TTS |
| VAD(端点检测) | silero-vad | ≥3.5.0 | 语音活动检测 |
| 向量数据库 | ChromaDB | ≥0.4.0 | 语义记忆向量检索 |
| 关系数据库 | SQLite3 | 内置 | 长期记忆持久化 |
| 缓存 | Redis（可选）或内存 | - | TTS语音缓存 |
| 音频处理 | numpy + soundfile / pydub | - | 音频格式转换 |

### 3.1 对话系统 (PetAgent)

#### 架构设计

```python
# 文件路径: app/agent/pet_agent.py

class PetAgent:
    """宠物人格AI Agent"""
    
    def __init__(self):
        self.config = get_config()
        self.llm_client = OpenAI(
            base_url=self.config.LLM_API_BASE_URL,
            api_key=self.config.LLM_API_KEY,
        )
        self.prompt_builder = PromptTemplateManager()
        self.memory_manager = MemoryManager()
        self.emotion_engine = EmotionEngine()
    
    async def chat(self, request: ChatRequest) -> ChatResponse:
        """
        完整对话流程：
        1. 组装上下文(人格+属性+性格+记忆+情绪)
        2. 渲染完整Prompt
        3. 调用LLM生成回复
        4. 后处理(提取关键信息→存入记忆)
        5. 分析回复情绪 → 更新宠物情绪
        6. 可选：生成TTS
        """
        start_time = time.time()
        
        try:
            # Step 1-2: 组装Prompt
            context = await self._build_context(request)
            prompt = self.prompt_builder.render_chat_prompt(context)
            
            # Step 3: 调用LLM
            reply_text = await self._call_llm(prompt)
            
            # Step 4: 记忆处理
            memory_extracted = await self.memory_manager.process_dialog_turn(
                user_message=request.message,
                pet_reply=reply_text,
                session_id=request.session_id
            )
            
            # Step 5: 情绪分析
            emotion_result = self.emotion_engine.analyze_text(reply_text)
            
            # Step 6: TTS(可选)
            audio_data = None
            audio_format = None
            audio_duration_ms = 0
            if request.need_voice:
                audio_data, audio_format, audio_duration_ms = \
                    await TTSEngine.get_instance().synthesize(reply_text)
            
            return ChatResponse(
                success=True,
                reply=reply_text,
                emotion=emotion_result.primary_emotion,
                memoryExtracted=memory_extracted,
                audioData=audio_data,
                audioFormat=audio_format or 'mp3',
                audioDurationMs=audio_duration_ms,
                processingTimeMs=int((time.time() - start_time) * 1000),
            )
            
        except Exception as e:
            logger.error(f"[PetAgent] Chat failed: {e}", exc_info=True)
            raise
```

2. 回复要简洁可爱（不超过50个字），适当使用语气词（喵、汪、~等）
3. 根据当前心情和精力调整语气：
   - 心情好时：活泼热情
   - 心情一般时：温和正常
   - 心情差时：略带委屈或不开心
   - 精力低时：慵懒困倦
4. 如果亲密度高，可以更亲密地称呼主人；亲密度低则保持礼貌距离
5. 根据毒舌值调整说话风格：高毒舌时可以适当调皮/怼人
"""


def render_chat_prompt(context: dict) -> str:
    """渲染完整的对话Prompt"""
    
    # 心情描述映射
    mood_desc_map = {
        range(0, 30): "很低落",
        range(30, 50): "有些郁闷",
        range(50, 70): "还不错",
        range(70, 85): "很开心",
        range(85, 101): "超级兴奋！",
    }
    
    def get_mood_desc(val):
        for r, desc in mood_desc_map.items():
            if val in r:
                return desc
        return "一般"
    
    # 亲密度描述
    intimacy_desc_map = {
        range(0, 20): "刚认识不久，还有点害羞",
        range(20, 50): "慢慢熟悉起来了",
        range(50, 80): "已经很亲近了",
        range(80, 101): "非常亲密，像家人一样",
    }
    
    def get_intimacy_desc(val):
        for r, desc in intimacy_desc_map.items():
            if val in r:
                return desc
        return ""
    
    # 精力描述
    energy_desc = "精力充沛" if context['energy'] > 60 else \
                  "有点累了" if context['energy'] > 30 else "快累趴了..."
    
    return SYSTEM_PROMPT_TEMPLATE.format(
        name=context['pet_name'],
        breed=context['pet_breed'],
        cheerfulness=context['personality'].get('cheerfulness', 50),
        naughtiness=context['personality'].get('naughtiness', 30),
        curiosity=context['personality'].get('curiosity', 50),
        independence=context['personality'].get('independence', 50),
        warmth=context['personality'].get('warmth', 50),
        mood=context['mood'],
        mood_desc=get_mood_desc(context['mood']),
        energy=context['energy'],
        energy_desc=energy_desc,
        emotion=context.get('emotion', 'calm'),
        intimacy=context['intimacy'],
        intimacy_desc=get_intimacy_desc(context['intimacy']),
        stage=context.get('stage', 'stranger'),
        user_profile=context.get('user_profile', '暂无'),
        recent_memories=context.get('recent_memories', '(无)'),
    )

### 3.2 记忆系统 (MemoryManager)

#### 三层记忆实现

```python
# 文件路径: app/memory/__init__.py

class MemoryManager:
    """三层记忆统一管理入口"""
    
    def __init__(self):
        self.short_term = ShortTermMemory()    # 内存队列
        self.semantic = SemanticMemory()       # ChromaDB
        self.long_term = LongTermMemory()      # SQLite
        self.compressor = MemoryCompressor()   # 压缩器
    
    async def process_dialog_turn(
        self, 
        user_message: str, 
        pet_reply: str, 
        session_id: str
    ) -> bool:
        """
        处理一轮对话的记忆操作：
        1. 存入短期记忆
        2. 检查是否需要压缩
        3. 提取关键信息存入语义/长期记忆
        """
        
        # 1. 存入短期记忆
        self.short_term.add(session_id, {
            'role': 'user',
            'content': user_message,
            'timestamp': time.time(),
        })
        self.short_term.add(session_id, {
            'role': 'pet',
            'content': pet_reply,
            'timestamp': time.time(),
        })
        
        # 2. 检查是否触发压缩
        turn_count = self.short_term.get_count(session_id)
        config = get_config()
        
        extracted = False
        if turn_count >= config.MEMORY_COMPRESS_THRESHOLD:
            # 执行压缩：摘要→语义记忆
            summary = await self.compressor.compress_session(session_id)
            await self.semantic.store(summary)
            extracted = True
        
        # 3. 尝试提取关键事实到长期记忆
        facts = await self._extract_facts(user_message, pet_reply)
        for fact in facts:
            await self.long_term.store(fact)
            extracted = True
        
        return extracted
    
    async def retrieve_context(self, session_id: str, query: str) -> str:
        """
        为新对话组装记忆上下文：
        1. 获取短期记忆(最近N轮原文)
        2. 语义搜索相关长期记忆
        3. 格式化为Prompt片段
        """
        recent = self short_term.get_recent(session_id, n=5)
        related = await self.semantic.search(query, limit=5)
        
        parts = []
        if related:
            parts.append("【相关回忆】")
            for item in related:
                parts.append(f"- {item.content} (相似度:{item.similarity:.2f})")
        
        return "\n".join(parts) if parts else "(无相关回忆)"
```

### 3.3 语音引擎 (VoiceEngine)

```python
# 文件路径: app/tts/edge_tts_engine.py

import asyncio
import edge_tts
import base64
import io


class EdgeTTSEngine:
    """Edge-TTS在线语音合成引擎"""
    
    _instance = None
    
    @classmethod
    def get_instance(cls) -> 'EdgeTTSEngine':
        if cls._instance is None:
            cls._instance = cls()
        return cls._instance
    
    async def synthesize(
        self, 
        text: str,
        voice: str = None,
        rate: str = None,
        pitch: str = None,
    ) -> tuple[bytes, str, int]:
        """
        合成语音
        @returns (audio_bytes, format, duration_ms)
        """
        config = get_config()
        voice = voice or config.TTS_DEFAULT_VOICE
        rate = rate or config.TTS_RATE
        pitch = pitch or config.TTS_PITCH
        
        communicate = edge_tts.Communicate(text, voice, rate=rate, pitch=pitch)
        
        # 流式收集音频数据
        audio_chunks = []
        async for chunk in communicate.stream():
            if chunk["type"] == "audio":
                audio_chunks.append(chunk["data"])
        
        audio_data = b''.join(audio_chunks)
        
        # Edge-TTS输出MP3格式
        # 估算时长(粗略): 约150字/分钟
        duration_estimate_ms = len(text) * 60000 // 150
        
        return audio_data, 'mp3', duration_estimate_ms


class WhisperEngine:
    """Whisper离线语音识别引擎"""
    
    _model = None
    _model_lock = asyncio.Lock()
    
    @classmethod
    async def get_model(cls):
        """懒加载模型(线程安全)"""
        if cls._model is None:
            async with cls._model_lock:
                if cls._model is None:  # double-check
                    import whisper
                    config = get_config()
                    logger.info(f"[Whisper] Loading model: {config.WHISPER_MODEL_SIZE}")
                    cls._model = whisper.load_model(
                        config.WHISPER_MODEL_SIZE,
                        device=config.WHISPER_DEVICE,
                    )
        return cls._model
    
    async def transcribe(
        self, 
        audio_bytes: bytes,
        language: str = None,
    ) -> STTResponse:
        """将音频转文字"""
        import numpy as np
        import io
        import soundfile as sf
        
        config = get_config()
        language = language or config.WHISPER_LANGUAGE
        
        model = await cls.get_model()
        
        # 将bytes转为numpy数组
        audio_buffer = io.BytesIO(audio_bytes)
        data, sr = sf.read(audio_buffer, dtype='float32')
        
        # 转录
        result = model.transcribe(data, language=language, fp16=False)
        
        return STTResponse(
            success=True,
            text=result["text"].strip(),
            confidence=max(result.get("avg_logprob", -1), 0),  # 归一化
            language=result["language"],
            duration_ms=int(len(data) / sr * 1000),
        )
```

# 文件路径: app/dream/generator.py

DREAM_PROMPT_TEMPLATE = """
你是一只名叫{pet_name}的虚拟宠物。
今天你的平均心情是{mood_avg}/100。
今天发生了这些事：
{key_events}

要求：

2. 用第一人称("我")叙述
3. 控制在50-100字之间
4. 要有趣味性和想象力
5. 给这个梦起一个标题(10字以内)

输出格式(严格JSON):
{{"title": "梦境标题", "content": "梦境内容描述"}}
"""

class DreamGenerator:
    """梦境生成器"""
    
    async def generate(
        self, 
        pet_name: str,
        daily_mood_avg: float,
        key_events: list[str],
        current_emotion: str,
    ) -> DreamRecord:
        # 1. 决定梦境类型
        dream_type = self._determine_dream_type(daily_mood_avg, current_emotion)
        
        # 2. 构建Prompt并调用LLM
        prompt = DREAM_PROMPT_TEMPLATE.format(
            pet_name=pet_name,
            mood_avg=daily_mood_avg,
            key_events="\n".join(f"- {e}" for e in key_events),
            emotion=current_emotion,
            dream_type=dream_type,
        )
        
        raw_response = await self._call_llm_for_json(prompt)
        
        # 3. 解析结果
        dream_data = json.loads(raw_response)
        
        # 4. 创建记录
        record = DreamRecord(
            dream_id=str(uuid.uuid4()),
            date=datetime.now().strftime("%Y-%m-%d"),
            dream_type=dream_type,
            title=dream_data["title"],
            content=dream_data["content"],
            emotion_before=current_emotion,
            created_at=int(time.time() * 1000),
        )
        
        # 5. 存储到SQLite
        await DreamJournalDAO.insert(record)
        
        # 6. 生成气泡文本
        bubble_texts = {
            "happy": f"主人！我做了一个好梦~{record.title}",
            "sad": f"呜呜...我做了一个噩梦...",
            "recall": f"主人，我梦到了以前的事...",
            "fantasy": f "哇！我做了一个超酷的梦！",
            "food": f "嘿嘿...梦里全是好吃的...",
        }
        record.bubble_text = bubble_texts.get(dream_type, "我做了一个梦~")
        
        return record
    
    def _determine_dream_type(self, mood_avg: float, emotion: str) -> str:
        rand = random.random()
        if mood_avg > 80 and rand < 0.6:
            return "happy"
        elif mood_avg < 30 and rand < 0.7:
            return "sad"
        elif rand < 0.15:
            return "fantasy"
        elif rand < 0.25:
            return "recall"
        else:
            return emotion  # 默认跟随当前情绪对应类型

### 6.2 北向端降级适配

当北向ArkTS客户端检测到AI服务不可用时，自动切换到以下策略：

| AI功能 | 正常 | 降级 | 实现方式 |
|--------|------|------|----------|
| 对话 | LLM实时生成 | `FALLBACK_RESPONSES`随机选取 | AIService内部try-catch + fallback池 |
| 语音识别 | Whisper | 禁用麦克风按钮 | UI置灰voice按钮 |
| 语音合成 | Edge-TTS | 显示纯文字气泡 | ChatPage判断audioData是否存在 |
| 记忆存储 | ChromaDB+SQLite | 仅RDB本地 | MemoryService降级写入RDB |
| 做梦 | LLM生成 | 跳过本次 | DreamService捕获异常后return null |
| 情绪分析 | LLM分析 | 基于属性值简单规则 | EmotionService.fallbackAnalyze() |
